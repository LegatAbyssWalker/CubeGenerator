World::World(GLuint blockAmount) {
	//Cube generation
	for (GLsizei x = 0; x < UNKNOWN_VARIABLE(VERTICES?); x++) {
        for (GLsizei z = 0; z < UNKNOWN_VARIABLE(VERTICES?); z++) {
            cubeVector.push_back(new CubeGenerator()); //Emplace back? Maybe even unique pointer for data holding? 
            
            cubeVector[x].setTexture(Texture::get(GRASS_TEXTURE_LOCATION));
            cubeVector[x].setPosition(glm::vec3(x, 0, z));
        }
	}
}
